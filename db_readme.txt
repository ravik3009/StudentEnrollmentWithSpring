CREATE TABLE Student (
   STUDENT_ID BIGINT NOT NULL GENERATED ALWAYS AS IDENTITY,
   DATEOFBIRTH timestamp,
   EMAILADDRESS VARCHAR(255),
   FIRSTNAME VARCHAR(255),
   LASTNAME VARCHAR(255),
   PASSWORD VARCHAR(20),
   USERNAME VARCHAR(20),
   PRIMARY KEY (STUDENT_ID)
);

DUMMY VARCHAR(255)
